1. Use Claude to generate realistic claims for the given company/ticker (since we're not running Whisper locally)

3. After all claims, compute a final score using Claude and stream it as `data-score`

Implementation using `generateText` from "ai" with the anthropic provider inside `createUIMessageStream`:

```typescript
import { createUIMessageStream, createUIMessageStreamResponse, generateText } from "ai";
import { anthropic } from "@ai-sdk/anthropic";
import type { AppUIMessage } from "@/lib/api-types";
import type { Claim, Score } from "@/lib/schemas";
import { fetchFinancials, fetchHistorical } from "@/lib/api-client";

export const runtime = "nodejs";
export const dynamic = "force-dynamic";

const sleep = (ms: number) => new Promise((r) => setTimeout(r, ms));

export async function POST(req: Request) {
  const body = await req.json().catch(() => ({}));
  const ticker = (body.ticker as string | undefined) ?? "AAPL";
  const companyName = (body.companyName as string | undefined) ?? ticker;

  const stream = createUIMessageStream<AppUIMessage>({
    execute: async ({ writer }) => {
      await sleep(300);

      // Step 1: Fetch context from Python backend (graceful fallback if offline)
      let financialContext = "";
      let historicalContext = "";
      try {
        const [fin, hist] = await Promise.all([
          fetchFinancials(ticker),
          fetchHistorical(ticker),
        ]);
        financialContext = `Revenue trend: ${fin.revenue_trend}. Margin trend: ${fin.margin_trend}. Data quality: ${fin.data_quality}.`;
        historicalContext = hist.data.map((d) => d.text.slice(0, 1000)).join("\n");
      } catch {
        financialContext = `No live financial data available for ${ticker}.`;
        historicalContext = "";
      }

      // Step 2: Ask Claude to extract realistic claims for this company's recent earnings call
      const claimsResult = await generateText({
        model: anthropic("claude-sonnet-4-5"),
        prompt: `You are analyzing a recent earnings call for ${companyName} (${ticker}).

Financial context: ${financialContext}

Generate 5-7 key forward-looking claims or promises that a ${companyName} executive would realistically make in a recent earnings call.

Return ONLY a JSON array, no markdown:
[
  {
    "id": "c1",
    "timestamp": 15,
    "text": "short summary (max 10 words)",
    "verbatim": "exact quote as executive would say it (1-2 sentences)"
  }
]

Use realistic timestamps between 10 and 300 seconds. Make claims specific and plausible for ${ticker}.`,
        maxTokens: 1500,
      });

      let claims: Claim[] = [];
      try {
        const raw = claimsResult.text.trim().replace(/```json\n?|\n?```/g, "");
        claims = JSON.parse(raw) as Claim[];
      } catch {
        // Fallback to a single placeholder claim
        claims = [{ id: "c1", timestamp: 30, text: "Analysis unavailable", verbatim: "Could not parse claims from AI response." }];
      }

      // Step 3: Stream claims with delays (simulating real-time discovery)
      for (const claim of claims) {
        await sleep(1500);
        writer.write({ type: "data-claim", id: claim.id, data: claim });
      }

      // Step 4: Ask Claude to score overall credibility
      await sleep(500);
      const scoreResult = await generateText({
        model: anthropic("claude-sonnet-4-5"),
        prompt: `Based on these earnings call claims from ${companyName} (${ticker}):
${claims.map((c) => `- ${c.text}`).join("\n")}

Historical context: ${historicalContext.slice(0, 2000) || "Not available."}
Financial context: ${financialContext}

Score this company's credibility. Return ONLY a JSON object:
{
  "pastDelivery": <0-100>,
  "currentMarket": <0-100>,
  "overall": "Likely" | "Mixed" | "Unlikely"
}

Be realistic. pastDelivery = how well this company has historically delivered on promises. currentMarket = how favorable current financials are for delivering these claims.`,
        maxTokens: 300,
      });

      let score: Score = { pastDelivery: 50, currentMarket: 50, futureSimulation: null, overall: "Mixed" };
      try {
        const raw = scoreResult.text.trim().replace(/```json\n?|\n?```/g, "");
        const parsed = JSON.parse(raw) as Partial<Score>;
        score = { ...score, ...parsed };
      } catch { /* use fallback */ }

      writer.write({ type: "data-score", id: "score-live", data: score });
    },
  });

  return createUIMessageStreamResponse({ stream });
}
```

```typescript
const sleep = (ms: number) => new Promise((r) => setTimeout(r, ms));

type Writer = Parameters<
  NonNullable<Parameters<typeof createUIMessageStream<AppUIMessage>>[0]["execute"]>
>[0]["writer"];

// Stream lines for a panel from an array of strings
async function streamLines(writer: Writer, panel: FeedPanel, lines: string[], baseDelay = 350) {
  for (let i = 0; i < lines.length; i++) {
    await sleep(baseDelay + Math.random() * 200);
    writer.write({
      type: "data-feed-line",
      id: `${panel}-${i}`,
      data: { panel, line: lines[i], done: i === lines.length - 1 },
    });
  }
}

// Stream a Claude response as individual lines to a panel
async function streamClaudeToPanel(writer: Writer, panel: FeedPanel, prompt: string, prefix?: string) {
  if (prefix) {
    writer.write({ type: "data-feed-line", id: `${panel}-prefix`, data: { panel, line: prefix, done: false } });
    await sleep(200);
  }

  const result = await generateText({
    model: anthropic("claude-sonnet-4-5"),
    prompt,
    maxTokens: 600,
  });

  // Split response into lines and stream them
  const lines = result.text
    .split("\n")
    .map((l) => l.trim())
    .filter(Boolean);

  for (let i = 0; i < lines.length; i++) {
    await sleep(300 + Math.random() * 150);
    writer.write({
      type: "data-feed-line",
      id: `${panel}-line-${i}`,
      data: { panel, line: lines[i], done: i === lines.length - 1 },
    });
  }
}
```
